630static void
631reload_mouse_modifiers (void)
632{
633 XModifierKeymap *modmap;
634 KeySym *keymap;
635 int keysyms_per_keycode;
636 int map_size;
637 int i;
638 gboolean have_meta;
639 gboolean have_hyper;
640 gboolean have_super;
641 int min_keycode, max_keycode;
642 int mod_meta, mod_super, mod_hyper;
643
644 XDisplayKeycodes (gdk_x11_display_get_xdisplay(gdk_display_get_default()),
645 &min_keycode,
646 &max_keycode);
647
648 keymap = XGetKeyboardMapping (gdk_x11_display_get_xdisplay(gdk_display_get_default()),
649 min_keycode,
650 max_keycode - min_keycode,
651 &keysyms_per_keycode);
652
653 modmap = XGetModifierMapping (gdk_x11_display_get_xdisplay(gdk_display_get_default()));
654
655 have_super = FALSE(0);
656 have_meta = FALSE(0);
657 have_hyper = FALSE(0);
658
659 /* there are 8 modifiers, and the first 3 are shift, shift lock,
660 * and control
661 */
662 map_size = 8 * modmap->max_keypermod;
663 i = 3 * modmap->max_keypermod;
664 mod_meta = mod_super = mod_hyper = 0;
665 while (i < map_size) {
666 /* get the key code at this point in the map,
667 * see if its keysym is one we're interested in
668 */
669 int keycode = modmap->modifiermap[i];
670
671 if (keycode >= min_keycode &&
672 keycode <= max_keycode) {
673 int j = 0;
674 KeySym *syms = keymap + (keycode - min_keycode) * keysyms_per_keycode;
675
676 while (j < keysyms_per_keycode) {
677 if (syms[j] == XK_Super_L0xffeb || syms[j] == XK_Super_R0xffec)
678 mod_super = i / modmap->max_keypermod;
679 else if (syms[j] == XK_Hyper_L0xffed || syms[j] == XK_Hyper_R0xffee)
680 mod_hyper = i / modmap->max_keypermod;
681 else if ((syms[j] == XK_Meta_L0xffe7 || syms[j] == XK_Meta_R0xffe8))
682 mod_meta = i / modmap->max_keypermod;
683 ++j;
684 }
685 }
686
687 ++i;
688 }
689
690 if ((1 << mod_meta) != Mod1Mask(1<<3))
691 have_meta = TRUE(!(0));
692 if (mod_super != 0 && mod_super != mod_meta)
693 have_super = TRUE(!(0));
694 if (mod_hyper != 0 && mod_hyper != mod_meta && mod_hyper != mod_super)
695 have_hyper = TRUE(!(0));
696
697 XFreeModifiermap (modmap);
698 XFree (keymap);
699
700 i = 0;
701 while (i < n_mouse_modifiers) {
702 g_free (mouse_modifiers[i].name);
703 if (mouse_modifiers[i].radio)
704 gtk_widget_destroy (mouse_modifiers[i].radio);
705 ++i;
706 }
707 g_free (mouse_modifiers);
708 mouse_modifiers = NULL((void*)0);
709
710 n_mouse_modifiers = 1; /* alt */
711 if (have_super)
712 ++n_mouse_modifiers;
713 if (have_hyper)
714 ++n_mouse_modifiers;
715 if (have_meta)
716 ++n_mouse_modifiers;
717

(MouseClickModifier)));

730 mouse_modifiers[i].value = "Hyper";

732 }

734 if (have_super) {

737 mouse_modifiers[i].value = "Super";
738 ++i;
